    <script>
        let selectedProfile = null;
        let editingProfileCard = null;

        function logout() {
            alert('Logged out!');
            // Redirect to login page or home
            window.location.href = 'login.html';
        }

        // Profile selection
        const profiles = document.querySelectorAll('.profile-card');
        profiles.forEach(profile => {
            profile.addEventListener('click', function(e) {
                // Prevent click if clicking on action buttons
                if (e.target.classList.contains('edit-btn') || e.target.classList.contains('remove-btn')) return;
                profiles.forEach(p => p.classList.remove('selected'));
                profile.classList.add('selected');
                selectedProfile = profile;
                // Example: Load different My List based on profile
                // document.getElementById('myListFrame').src = `myList.html?profile=${profile.getAttribute('data-profile')}`;
            });
        });

        // Edit Profile Modal
        const editProfileModal = document.getElementById('editProfileModal');
        const editProfileName = document.getElementById('editProfileName');
        const saveProfileBtn = document.getElementById('saveProfileBtn');
        const cancelEditBtn = document.getElementById('cancelEditBtn');

        document.querySelectorAll('.edit-btn').forEach(btn => {
            btn.addEventListener('click', function(e) {
                e.stopPropagation();
                editingProfileCard = btn.closest('.profile-card');
                editProfileName.value = editingProfileCard.querySelector('.profile-name').textContent;
                editProfileModal.classList.add('active');
                editProfileName.focus();
            });
        });

        saveProfileBtn.addEventListener('click', function() {
            if (editingProfileCard && editProfileName.value.trim() !== '') {
                editingProfileCard.querySelector('.profile-name').textContent = editProfileName.value.trim();
            }
            editProfileModal.classList.remove('active');
        });

        cancelEditBtn.addEventListener('click', function() {
            editProfileModal.classList.remove('active');
        });

        // Remove Profile
        document.querySelectorAll('.remove-btn').forEach(btn => {
            btn.addEventListener('click', function(e) {
                e.stopPropagation();
                const card = btn.closest('.profile-card');
                if (confirm('Are you sure you want to remove this profile?')) {
                    card.parentElement.remove();
                }
            });
        });

        // Add Profile
        document.getElementById('addProfileBtn').addEventListener('click', function() {
            const name = prompt('Enter new profile name:');
            if (name && name.trim() !== '') {
                const newProfileHTML = `
                    <div class="profile-card" data-profile="${name}">
                        <div class="profile-actions">
                            <button class="edit-btn">Edit</button>
                            <button class="remove-btn">Remove</button>
                        </div>
                        <img src="https://i.imgur.com/8Km9tLL.png" alt="${name}" class="profile-img">
                        <div class="profile-name">${name}</div>
                    </div>
                `;
                const col = document.createElement('div');
                col.className = 'col-6 col-sm-4 col-md-3';
                col.innerHTML = newProfileHTML;
                // Insert before the add profile card
                const addProfileCol = document.getElementById('addProfileBtn').parentElement;
                addProfileCol.parentElement.insertBefore(col, addProfileCol);

                // Add event listeners to new profile
                const newProfileCard = col.querySelector('.profile-card');
                newProfileCard.addEventListener('click', function(e) {
                    if (e.target.classList.contains('edit-btn') || e.target.classList.contains('remove-btn')) return;
                    document.querySelectorAll('.profile-card').forEach(p => p.classList.remove('selected'));
                    newProfileCard.classList.add('selected');
                    selectedProfile = newProfileCard;
                });
                col.querySelector('.edit-btn').addEventListener('click', function(e) {
                    e.stopPropagation();
                    editingProfileCard = newProfileCard;
                    editProfileName.value = newProfileCard.querySelector('.profile-name').textContent;
                    editProfileModal.classList.add('active');
                    editProfileName.focus();
                });
                col.querySelector('.remove-btn').addEventListener('click', function(e) {
                    e.stopPropagation();
                    if (confirm('Are you sure you want to remove this profile?')) {
                        col.remove();
                    }
                });
            }
        });

        // Close modal on outside click
        editProfileModal.addEventListener('click', function(e) {
            if (e.target === editProfileModal) {
                editProfileModal.classList.remove('active');
            }
        });
    </script>
